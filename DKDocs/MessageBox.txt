//MessageBox's



// ShowMessage (no return info)
////////////////////////////
if(Id(event, "ElementID")){
	SendEvent("DKMessage.html", "ShowMessage", "GetType(event),ElementID,Message");
}





// Confirm (ok, cancel)
///////////////////////////
if(Id(event, "ElementID")){
	if(!GetValue(event)){ SendEvent("DKMessage.html", "Confirm", "ElementID,GetType(event),Message"); return; }
	if(GetValue(event) != "OK"){ return; }
		
	//OK, commence 
}





// GetInput (return string, cancel)
//////////////////////////////
if(Id(event, "ElementID")){
	if(!GetValue(event)){ SendEvent("DKMessage.html", "GetInput", "ElementID,GetType(event),Message"); return; }
	if(GetValue(event) == "CANCEL"){ return; }
		
	//UseVariable(value);
}
